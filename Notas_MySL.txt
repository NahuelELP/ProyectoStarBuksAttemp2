A LEFT JOIN B
"Quiero todo A y lo que coincida de B"
A RIGHT JOIN B
"Quiero todo B y lo que coincida de A"

INNER JOIN
--Solo devuelve las coincidencias
SELECT clientes.nombre, pedidos.producto
FROM clientes
INNER JOIN pedidos
ON clientes.id = pedidos.cliente_id;

CROSS JOIN
-- Convina cada fila de la tabla de A con cada fila de la tabla de B
--Ej: tener tabla de colores y una tabla de talles, asi se tiene todos
los talles de color 

SELECT *
FROM colores
CROSS JOIN tallas;

GROUP BY
-- agrupa filas ke tengan el mismo valor en una columna
